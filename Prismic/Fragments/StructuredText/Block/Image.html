def as_html(link_resolver, html_serializer = nil)
  custom = nil
  unless html_serializer.nil?
    custom = html_serializer.serialize(self, '')
  end
  if custom.nil?
    classes = ['block-img']
    unless @label.nil?
      classes.push(@label)
    end
    %(<p class="#{classes.join(' ')}">#{view.as_html(link_resolver)}</p>)
  else
    custom
  end
end
